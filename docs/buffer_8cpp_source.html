84 void BufMgr::allocBuf(FrameId &frame)
85 {
86 // find a frame to allocate
87
88 bool foundUnpin = false;
89 FrameId start = clockHand;
90 // printf("start: %d\n", start);
91 while (1)
92 {
93 advanceClock();
94
95 if (bufDescTable[clockHand].valid == false)
96 {
97 // printf("found unpinned buffer: %d\n", clockHand);
98 frame = clockHand;
99 return;
100 }
101 else if (bufDescTable[clockHand].pinCnt == 0)
102 {
103 foundUnpin = true;
104 if (bufDescTable[clockHand].refbit == false)
105 {
106 // if the frame is dirty, write it back to disk
107 if (bufDescTable[clockHand].dirty == true)
108 {
109 // bufDescTable[clockHand].file->writePage(bufDescTable[clockHand].pageNo, bufPool[clockHand]);
110 bufDescTable[clockHand].file->writePage(bufPool[clockHand]);
111 hashTable->remove(bufDescTable[clockHand].file, bufDescTable[clockHand].pageNo);
112 bufDescTable[clockHand].Clear();
113 }
114 // printf("found unpinned buffer: %d\n", clockHand);
115 frame = clockHand;
116 return;
117 }
118 }
119 bufDescTable[clockHand].refbit = false;
120 if (clockHand == start) // if we've looped around
121 {
122 if (foundUnpin == false)
123 {
124 // printf("ERROR: no unpinned buffers available with clockhand: %d\n", clockHand);
125 throw BufferExceededException();
126 }
127 else
128 foundUnpin = false;
129 }
130 }

143 void BufMgr::readPage(File *file, const PageId pageNo, Page *&page)
144 {
145 FrameId frame;
146 try
147 {
148 hashTable->lookup(file, pageNo, frame);
149
150 // page is in buffer pool
151 bufDescTable[frame].refbit = true;
152 bufDescTable[frame].pinCnt++;
153 page = &bufPool[frame];
154 }
155 catch (HashNotFoundException e)
156 {
157
158 // page not in buffer pool
159 allocBuf(frame);
160 bufPool[frame] = file->readPage(pageNo);
161 bufDescTable[frame].Set(file, pageNo);
162 hashTable->insert(file, pageNo, frame);
163 page = &bufPool[frame];
164 }
