54int main()
55{
56 //Following code shows how to you File and Page classes
57
58 const std::string& filename = "test.db";
59 // Clean up from any previous runs that crashed.
60 try
61 {
62 File::remove(filename);
63 }
64 catch(FileNotFoundException)
65 {
66 }
67
68 {
69 // Create a new database file.
70 File new_file = File::create(filename);
71
72 // Allocate some pages and put data on them.
73 PageId third_page_number;
74 for (int i = 0; i < 5; ++i) {
75 Page new_page = new_file.allocatePage();
76 if (i == 3) {
77 // Keep track of the identifier for the third page so we can read it
78 // later.
79 third_page_number = new_page.page_number();
80 }
81 new_page.insertRecord("hello!");
82 // Write the page back to the file (with the new data).
83 new_file.writePage(new_page);
84 }
85
86 // Iterate through all pages in the file.
87 for (FileIterator iter = new_file.begin();
88 iter != new_file.end();
89 ++iter) {
90 // Iterate through all records on the page.
91 for (PageIterator page_iter = (*iter).begin();
92 page_iter != (*iter).end();
93 ++page_iter) {
94 std::cout << "Found record: " << *page_iter
95 << " on page " << (*iter).page_number() << "\n";
96 }
97 }
98
99 // Retrieve the third page and add another record to it.
100 Page third_page = new_file.readPage(third_page_number);
101 const RecordId& rid = third_page.insertRecord("world!");
102 new_file.writePage(third_page);
103
104 // Retrieve the record we just added to the third page.
105 std::cout << "Third page has a new record: "
106 << third_page.getRecord(rid) << "\n\n";
107 }
108 // new_file goes out of scope here, so file is automatically closed.
109
110 // Delete the file since we're done with it.
111 File::remove(filename);
112
113 //This function tests buffer manager, comment this line if you don't wish to test buffer manager
114 testBufMgr();
115}
